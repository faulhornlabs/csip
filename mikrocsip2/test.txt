id : Pi U (A. A -> A)
  = A. x. ret x;

f : U * U -> U * U
  = p. pair p (A. B. ret (B , A));

bool = record Top (_. Top + Top);
Bool  : U    =  ret (bool tt);
False : Bool =  ret (wrap (Left  tt));
True  : Bool =  ret (wrap (Right tt));

and : Bool -> Bool -> Bool
  = a. b. unwrap a (a. either a (_. ret False) (_. ret b));

ite : Pi U (A. Bool -> A -> A -> A)
  = A. b. t. f. unwrap b (b. either b (_. ret f) (_. ret t));

Maybe = record U (A. Top + A);
Nothing : Pi U Maybe              = A. ret (wrap (Left tt));
Just    : Pi U (A. A -> Maybe A)  = A. a. ret (wrap (Right a));

f (id U Top , ite U True Bot U) : id U (U * U)
